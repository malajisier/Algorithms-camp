LFU的实现   
思路：
- 时间复杂度要求O(1)，类似于LRU，但需要维护一个 key的使用频次
- 一个哈希表存频次，以频次freq 作为key索引，按照key使用顺序存到双向链表中
- 另一个哈希表存key所在的内存地址，key作为索引，val 为key在freqMap中 链表里的内存地址


### 方法一：双哈希表+双向链表
```java
class LFUCache {
    Map<Integer, Node> cache;  // key为索引，val存放对应在freqMap中链表的内存地址
    Map<Integer, DoubleLinkedList> freqMap;
    int size;
    int cap;
    int minFreq;  // 最小频次，便于更新

    public LFUCache(int capacity) {
        cache = new HashMap<>();
        freqMap = new HashMap<>();
        this.cap = capacity;
    }
    
    public int get(int key) {
        Node node = cache.get(key);
        if (node == null) return -1;
        updateFreq(node);  // 更新node的位置
        return node.val;
    }
    
    public void put(int key, int val) {
        if (cap == 0) return;
        Node node = cache.get(key);
        if (node != null) {
            node.val = val;
            updateFreq(node);
        } else {
            if (size == cap) {
                DoubleLinkedList minFreqList = freqMap.get(minFreq);
                Node lastNode = minFreqList.tail.prev;
                cache.remove(lastNode.key);
                minFreqList.removeNode(lastNode);
                size--;
            }
            Node newNode = new Node(key, val);
            cache.put(key, newNode);
            DoubleLinkedList oneFreqList = freqMap.get(1);
            if (oneFreqList == null) {
                oneFreqList = new DoubleLinkedList();
                freqMap.put(1, oneFreqList);
            }
            oneFreqList.addFirst(newNode);
            size++;
            minFreq = 1;
        }
    }

    public void updateFreq(Node node) {
        int freq = node.freq;
        DoubleLinkedList list = freqMap.get(freq);
        list.removeNode(node);
        if (freq == minFreq && list.head.next == list.tail) {
            minFreq = freq + 1;
        }

        node.freq++;
        list = freqMap.get(freq + 1);
        if (list == null) {
            list = new DoubleLinkedList();
            freqMap.put(freq + 1, list);
        }
        list.addFirst(node); 
    }
}


class Node {
    int key;
    int val;
    int freq = 1;
    Node prev;
    Node next;

    public Node() {}
    public Node(int key, int val) {
        this.key = key;
        this.val = val;
    }
}

class DoubleLinkedList {
    Node head;
    Node tail;

    public DoubleLinkedList() {
        head = new Node();
        tail = new Node();
        head.next = tail;
        tail.prev = head;
    }

    public void removeNode(Node node) {
        node.prev.next = node.next;
        node.next.prev = node.prev;
    }

    public void addFirst(Node node) {
        node.next = head.next;
        node.prev = head;
        head.next.prev = node;
        head.next = node;
    }
}
```
